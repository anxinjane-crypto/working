Google Sheets 設定指南

步驟 1: 建立 Google Sheet

建立一個新的 Google Sheet，包含以下工作表(Sheets)：

  • Config - 系統設定
  • Categories - 分類資料
  • Items - 打卡項目
  • CheckInLog - 打卡紀錄

步驟 2: 設定 Google Apps Script

在 Google Sheet 中，點擊「擴充功能」>「Apps Script」，貼上以下程式碼：

// Google Apps Script 程式碼
const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();
const SUPERVISOR_PASSWORD = 'admin123'; // 請修改主管密碼

function doGet(e) {
return handleRequest(e);
}

function doPost(e) {
return handleRequest(e);
}

function handleRequest(e) {
const action = e.parameter.action;
let result;

try {
  switch(action) {
    case 'getData': result = getData(); break;
    case 'checkIn': result = checkIn(e.parameter); break;
    case 'addItem': result = addItem(e.parameter); break;
    case 'updateItem': result = updateItem(e.parameter); break;
    case 'deleteItem': result = deleteItem(e.parameter); break;
    case 'supervisorCheck': result = supervisorCheck(e.parameter); break;
    case 'addCategory': result = addCategory(e.parameter); break;
    case 'addSubCategory': result = addSubCategory(e.parameter); break;
    case 'getLog': result = getLog(e.parameter); break;
    default: result = {error: 'Unknown action'};
  }
} catch(err) {
  result = {error: err.toString()};
}

return ContentService.createTextOutput(JSON.stringify(result))
  .setMimeType(ContentService.MimeType.JSON);
}

function checkIn(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const itemSheet = ss.getSheetByName('Items');
const logSheet = ss.getSheetByName('CheckInLog');
const itemId = parseInt(params.itemId);
const user = params.user || 'Anonymous';
const now = new Date().toLocaleString('zh-TW');

const data = itemSheet.getDataRange().getValues();
for (let i = 1; i < data.length; i++) {
  if (data[i][0] == itemId) {
    itemSheet.getRange(i + 1, 7).setValue(now + ' by ' + user);
    logSheet.appendRow([now, itemId, user, 'Check-in', '']);
    break;
  }
}
return {success: true, timestamp: now, user: user};
}

function addItem(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const itemSheet = ss.getSheetByName('Items');
const data = itemSheet.getDataRange().getValues();
const newId = data.length > 1 ? Math.max(...data.slice(1).map(r => r[0])) + 1 : 1;

itemSheet.appendRow([
  newId,
  params.categoryKey,
  params.name,
  params.responsible || '',
  params.frequency || '',
  params.notes || '',
  '',
  'false',
  ''
]);

return {success: true, id: newId};
}

function updateItem(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const itemSheet = ss.getSheetByName('Items');
const itemId = parseInt(params.itemId);
const field = params.field;
const value = params.value;

const data = itemSheet.getDataRange().getValues();
const headers = data[0];
const colIndex = headers.indexOf(field);

if (colIndex === -1) return {success: false, error: 'Field not found'};

for (let i = 1; i < data.length; i++) {
  if (data[i][0] == itemId) {
    itemSheet.getRange(i + 1, colIndex + 1).setValue(value);
    break;
  }
}
return {success: true};
}

function deleteItem(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const itemSheet = ss.getSheetByName('Items');
const itemId = parseInt(params.itemId);

const data = itemSheet.getDataRange().getValues();
for (let i = 1; i < data.length; i++) {
  if (data[i][0] == itemId) {
    itemSheet.deleteRow(i + 1);
    break;
  }
}
return {success: true};
}

function supervisorCheck(params) {
if (params.password !== SUPERVISOR_PASSWORD) {
  return {success: false, error: 'Wrong password'};
}

const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const itemSheet = ss.getSheetByName('Items');
const logSheet = ss.getSheetByName('CheckInLog');
const itemId = parseInt(params.itemId);
const now = new Date().toLocaleString('zh-TW');

const data = itemSheet.getDataRange().getValues();
for (let i = 1; i < data.length; i++) {
  if (data[i][0] == itemId) {
    const currentCheck = data[i][7] === 'true' || data[i][7] === true;
    itemSheet.getRange(i + 1, 8).setValue(!currentCheck ? 'true' : 'false');
    itemSheet.getRange(i + 1, 9).setValue(!currentCheck ? '已檢查 ' + now : '');
    logSheet.appendRow([now, itemId, 'Supervisor', !currentCheck ? 'Approved' : 'Unchecked', '']);
    break;
  }
}
return {success: true};
}

function addCategory(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const catSheet = ss.getSheetByName('Categories');
catSheet.appendRow([params.id, '', params.zh, params.en, 'main']);
return {success: true};
}

function addSubCategory(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const catSheet = ss.getSheetByName('Categories');
catSheet.appendRow([params.id, params.parentId, params.zh, params.en, 'sub']);
return {success: true};
}

function getLog(params) {
const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
const logSheet = ss.getSheetByName('CheckInLog');
const itemId = params.itemId;

const data = logSheet.getDataRange().getValues();
const logs = data.slice(1).filter(row => row[1] == itemId);
return {success: true, logs: logs};
}

步驟 3: 部署為網頁應用程式

  1. 在 Apps Script 編輯器中，點擊「部署」>「新增部署作業」
  2. 選擇類型為「網頁應用程式」
  3. 執行身分選擇「我」
  4. 誰可以存取選擇「所有人」
  5. 點擊「部署」並複製產生的網址
